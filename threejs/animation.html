Animate Single Cube

const clock = new THREE.Clock()

function animate() {
  const elapsedTime = clock.getElapsedTime();
  cube.rotation.y = elapsedTime;
  renderer.render(scene, camera);
  window.requestAnimationFrame(animate);
}
